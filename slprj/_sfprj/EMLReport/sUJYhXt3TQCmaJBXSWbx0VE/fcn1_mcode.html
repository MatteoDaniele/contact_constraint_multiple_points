1function y = fcn(u)
  2%% MATTEO DANIELE
  3% trying to create a contact constraint on platform that depends on the
  4% difference in position, speed, orientation and angular velocity between
  5% ground and "flying object"
  6mem_alloc = 500;
  7
  8GND = [u(1);u(2);u(3)];
  9GND_Phi = [u(4);u(5);u(6)];
 10% position of the point P ("flying object")
 11P = [u(7);u(8);u(9)];
 12% rotation matrix to define all vertices
 13R = eul2rotm([GND_Phi(3) GND_Phi(2) GND_Phi(1)]);
 14
 15% dimension of the platform along x OR 
 16% radius of the inscribed surface
 17Length_x = u(10);
 18% dimension of the platform along y
 19Length_y = u(11);
 20% vector of the planar distance between contact references
 21% the platform can be constitued of an arbitrary number of points (minimum
 22% number is 3 in order to have at least triangulation effectiveness
 23n_vertices = u(12);
 24% if vertices number is more than 4 then Length_x is the radius of the
 25% circumference inside which the surface is defined
 26alpha_vert = 2*pi/n_vertices;
 27% initialization of the vertices
 28GND_vert = zeros(3,mem_alloc);
 29% return from memory allocation to original sizes
 30GND_vert = GND_vert(:,1:n_vertices);
 31% initialization of normals and cosine vectors
 32z_normals = zeros(size(GND_vert));
 33ct = zeros(size(GND_vert));
 34
 35
 36
 37
 38% distance vector between point and ground
 39PG = P-GND;
 40% PG_unit = PG/norm(PG);
 41
 42if Length_y > 0
 43    % rectangle of base Length_x and height Length_y
 44    GND_vert = [ Length_x -Length_x -Length_x  Length_x
 45                 Length_y  Length_y -Length_y -Length_y
 46                    0           0       0       0       ];
 47               
 48    for k = 1:n_vertices
 49        GND_vert(:,k) = GND + R*GND_vert(:,k);
 50%             % distance vector between point and ground
 51%             PG(:,k) = P-GND_vert(:,k);
 52%             % normalization of the vector
 53%             PG(:,k) = PG(:,k)/norm(PG(:,k));
 54    end
 55    
 56else
 57    % polygon inscribed in circumference of radius Length_x
 58    Length_Vector = [Length_x;0;0];
 59
 60
 61    for k = 1:n_vertices
 62    R_vector = [cos((k-1)*alpha_vert) -sin((k-1)*alpha_vert) 0
 63                sin((k-1)*alpha_vert)  cos((k-1)*alpha_vert) 0
 64                            0                   0          1];
 65
 66    % definition of the vertex in the plane (oriented as necessary)
 67    GND_vert(:,k) = GND + R*R_vector*Length_Vector;
 68
 69%     PG(:,k) = P-GND_vert(:,k);
 70%     % normalization of the vector
 71%     PG(:,k) = PG(:,k)/norm(PG(:,k));
 72
 73    end
 74
 75end
 76
 77% distance vector between point and ground
 78PG = P-GND_vert;
 79for k = 1:n_vertices
 80% normalization of the vector
 81PG(:,k) = PG(:,k)/norm(PG(:,k));
 82end
 83
 84
 85
 86% normals to planes
 87for k = 1:n_vertices-1
 88    z_normals(:,k) = cross(PG(:,k),PG(:,k+1));
 89    z_normals(:,k) = z_normals(:,k)/norm(z_normals(:,k));
 90end
 91z_normals(:,end) = cross(PG(:,end),PG(:,1));
 92z_normals(:,end) = z_normals(:,end)/norm(z_normals(:,end));
 93
 94% platform plane normal (reference for cosine calculations)
 95z = cross(GND_vert(:,1)-GND,GND_vert(:,2)-GND);
 96z = z/norm(z);
 97
 98
 99
100
101% cosine of the angle between z and zij
102for k = 1:n_vertices
103    ct(k) = z_normals(:,k)'*z;
104end
105
106% all cosines close to unit means that contact happens
107ctflag = sum(ct);
108
109% initialize output
110y = zeros(4,1);
111
112% ctflag close to 4 means that contact happens
113% ideally when ctflag is 4 then contact happens, in reality ctflag must be
114% over (or under) a certain limit to activate forces (eg: ctflag > 3.9)
115y(1) = ctflag;
116% pass also the distance vector between "flying object" and ground
117% reference point
118y(2:end) = PG;
