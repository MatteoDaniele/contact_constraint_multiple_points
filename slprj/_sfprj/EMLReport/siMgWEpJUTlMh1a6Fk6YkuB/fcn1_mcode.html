1function y = fcn(u)
  2%% MATTEO DANIELE
  3% trying to create a contact constraint on platform that depends on the
  4% difference in position, speed, orientation and angular velocity between
  5% ground and "flying object"
  6mem_alloc = 500;
  7
  8% dynamic memory allocation bound
  9assert(u(12)<128);
 10
 11GND = [u(1);u(2);u(3)];
 12GND_Phi = [u(4);u(5);u(6)];
 13% position of the point P ("flying object")
 14P = [u(7);u(8);u(9)];
 15% rotation matrix to define all vertices
 16R = eul2rotm([GND_Phi(3) GND_Phi(2) GND_Phi(1)]);
 17
 18% dimension of the platform along x OR 
 19% radius of the inscribed surface
 20Length_x = u(10);
 21% dimension of the platform along y
 22Length_y = u(11);
 23% vector of the planar distance between contact references
 24% the platform can be constitued of an arbitrary number of points (minimum
 25% number is 3 in order to have at least triangulation effectiveness
 26n_vertices = u(12);
 27% if vertices number is more than 4 then Length_x is the radius of the
 28% circumference inside which the surface is defined
 29alpha_vert = 2*pi/n_vertices;
 30% initialization of the vertices
 31GND_vert = zeros(3,mem_alloc);
 32% return from memory allocation to original sizes
 33GND_vert = GND_vert(:,1:n_vertices);
 34% initialization of normals and cosine vectors
 35z_normals = zeros(size(GND_vert));
 36ct = zeros(size(GND_vert));
 37
 38
 39
 40
 41% distance vector between point and ground
 42PG = P-GND;
 43% PG_unit = PG/norm(PG);
 44
 45if Length_y > 0
 46    % rectangle of base Length_x and height Length_y
 47    GND_vert = [ Length_x -Length_x -Length_x  Length_x
 48                 Length_y  Length_y -Length_y -Length_y
 49                    0           0       0       0       ];
 50               
 51    for k = 1:n_vertices
 52        GND_vert(:,k) = GND + R*GND_vert(:,k);
 53            % distance vector between point and ground
 54            PG(:,k) = P-GND_vert(:,k);
 55            % normalization of the vector
 56            PG(:,k) = PG(:,k)/norm(PG(:,k));
 57    end
 58    
 59else
 60    % polygon inscribed in circumference of radius Length_x
 61    Length_Vector = [Length_x;0;0];
 62
 63
 64    for k = 1:n_vertices
 65    R_vector = [cos((k-1)*alpha_vert) -sin((k-1)*alpha_vert) 0
 66                sin((k-1)*alpha_vert)  cos((k-1)*alpha_vert) 0
 67                            0                   0          1];
 68
 69    % definition of the vertex in the plane (oriented as necessary)
 70    GND_vert(:,k) = GND + R*R_vector*Length_Vector;
 71
 72    PG(:,k) = P-GND_vert(:,k);
 73    % normalization of the vector
 74    PG(:,k) = PG(:,k)/norm(PG(:,k));
 75
 76    end
 77
 78end
 79
 80
 81
 82
 83% normals to planes
 84for k = 1:n_vertices-1
 85    z_normals(:,k) = cross(PG(:,k),PG(:,k+1));
 86    z_normals(:,k) = z_normals(:,k)/norm(z_normals(:,k));
 87end
 88z_normals(:,end) = cross(PG(:,end),PG(:,1));
 89z_normals(:,end) = z_normals(:,end)/norm(z_normals(:,end));
 90
 91% platform plane normal (reference for cosine calculations)
 92z = cross(GND_vert(:,1)-GND,GND_vert(:,2)-GND);
 93z = z/norm(z);
 94
 95
 96
 97
 98% cosine of the angle between z and zij
 99for k = 1:n_vertices
100    ct(k) = z_normals(:,k)'*z;
101end
102
103% all cosines close to unit means that contact happens
104ctflag = sum(ct);
105
106% initialize output
107y = zeros(4,1);
108
109% ctflag close to 4 means that contact happens
110% ideally when ctflag is 4 then contact happens, in reality ctflag must be
111% over (or under) a certain limit to activate forces (eg: ctflag > 3.9)
112y(1) = ctflag;
113% pass also the distance vector between "flying object" and ground
114% reference point
115y(2:end) = PG;
116
