1function y = fcn(u)
  2%% MATTEO DANIELE
  3% trying to create a contact constraint on platform that depends on the
  4% difference in position, speed, orientation and angular velocity between
  5% ground and "flying object"
  6
  7
  8GND = u(1:3)';
  9GND_Phi = u(4:6);
 10% position of the point P ("flying object")
 11P = u(7:9);
 12% rotation matrix to define all vertices
 13R = eul2rotm([GND_Phi(3) GND_Phi(2) GND_Phi(1)]);
 14
 15% dimension of the platform along x OR 
 16% radius of the inscribed surface
 17Length_x = u(10);
 18% dimension of the platform along y
 19Length_y = u(11);
 20% vector of the planar distance between contact references
 21% the platform can be constitued of an arbitrary number of points (minimum
 22% number is 3 in order to have at least triangulation effectiveness
 23n_vertices = u(12);
 24% if vertices number is more than 4 then Length_x is the radius of the
 25% circumference inside which the surface is defined
 26alpha_vert = 2*pi/n_vertices;
 27% initialization of the vertices
 28GND_vert = zeros(3,n_vertices);
 29
 30
 31% distance vector between point and ground
 32PG = P-GND;
 33% PG_unit = PG/norm(PG);
 34
 35if Length_y > 0
 36    % rectangle of base Length_x and height Length_y
 37    GND_vert = [ Length_x -Length_x -Length_x  Length_x
 38                 Length_y  Length_y -Length_y -Length_y
 39                    0           0       0       0       ];
 40               
 41    for k = 1:n_vertices
 42        GND_vert(:,k) = GND + R*GND_vert(:,k);
 43            % distance vector between point and ground
 44            PG(:,k) = P-GND_vert(:,k);
 45            % normalization of the vector
 46            PG(:,k) = PG(:,k)/norm(PG(:,k));
 47    end
 48    
 49else
 50    % polygon inscribed in circumference of radius Length_x
 51    Length_Vector = [Length_x;0;0];
 52
 53
 54    for k = 1:n_vertices
 55    R_vector = [cos((k-1)*alpha_vert) -sin((k-1)*alpha_vert) 0
 56                sin((k-1)*alpha_vert)  cos((k-1)*alpha_vert) 0
 57                            0                   0          1];
 58
 59    % definition of the vertex in the plane (oriented as necessary)
 60    GND_vert(:,k) = GND + R*R_vector*Length_Vector;
 61
 62    PG(:,k) = P-GND_vert(:,k);
 63    % normalization of the vector
 64    PG(:,k) = PG(:,k)/norm(PG(:,k));
 65
 66    end
 67
 68end
 69
 70% initialization of normals to planes formed by distance vectors
 71z_normals = zeros(3,n_vertices);
 72% normals to planes
 73for k = 1:n_vertices-1
 74    z_normals(:,k) = cross(PG(:,k),PG(:,k+1));
 75    z_normals(:,k) = z_normals(:,k)/norm(z_normals(:,k));
 76end
 77z_normals(:,end) = cross(PG(:,end),PG(:,1));
 78z_normals(:,end) = z_normals(:,end)/norm(z_normals(:,end));
 79
 80% platform plane normal (reference for cosine calculations)
 81z = cross(GND_vert(:,1)-GND,GND_vert(:,2)-GND);
 82z = z/norm(z);
 83
 84
 85% cosine of the angle between z and zij
 86ct = zeros(1,n_vertices);
 87for k = 1:n_vertices
 88    ct(k) = z_normals(:,k)'*z;
 89end
 90
 91% all cosines close to unit means that contact happens
 92ctflag = sum(ct);
 93
 94% initialize output
 95y = zeros(4,1);
 96
 97% ctflag close to 4 means that contact happens
 98% ideally when ctflag is 4 then contact happens, in reality ctflag must be
 99% over (or under) a certain limit to activate forces (eg: ctflag > 3.9)
100y(1) = ctflag;
101% pass also the distance vector between "flying object" and ground
102% reference point
103y(2:end) = PG;
